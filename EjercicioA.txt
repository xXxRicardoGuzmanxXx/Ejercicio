#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
void invertir(char *cad) {
    int i, j;
    char temp;
    int length = strlen(cad);
    for (i = 0, j = length - 1; i < j; i++, j--) {
        temp = cad[i];
        cad[i] = cad[j];
        cad[j] = temp;
    }
}
void contar(const char *cad, int *Palabras, int *Vocales, int *Consonantes, int *Espacios) {
    *Palabras = 0;
    *Vocales = 0;
    *Consonantes = 0;
    *Espacios = 0;
    int length = strlen(cad);
    int i;
    int enPalabra = 0; 
    for (i = 0; i < length; i++) {
        if (cad[i] == ' ') {
            *Espacios += 1;
            enPalabra = 0;
        } else {
            if (!enPalabra) {
                *Palabras += 1;
                enPalabra = 1;
            }
            char c = tolower(cad[i]);
            if (c == 'a' || c == 'e' || c == 'i' || c == 'o' || c == 'u')
                *Vocales += 1;
            else if (isalpha(c))
                *Consonantes += 1;
        }
    }
}
int main() {
    int n;
    printf("Â¿Cuantas cadenas desea ingresar? ");
    scanf("%d", &n);
    getchar();
    FILE *archivo = fopen("resultado.txt", "w");
    if (archivo == NULL) {
        printf("No se pudo abrir el archivo.\n");
        return 1;
    }
    int i;
    for (i = 0; i < n; i++) {
        char *cad = NULL;
        size_t capacidad = 0;
        printf("Ingrese su cadena: ");
        getline(&cad, &capacidad, stdin);
        cad[strcspn(cad, "\n")] = '\0';
        invertir(cad);
        int Palabras, Vocales, Consonantes, Espacios;
        contar(cad, &Palabras, &Vocales, &Consonantes, &Espacios);
        fprintf(archivo, "%s\n", cad);
        fprintf(archivo, "Tiene %d espacios, %d vocales y %d consonantes.\n", Espacios, Vocales, Consonantes);
        fprintf(archivo, "La cadena contiene %d palabras.\n", Palabras);
    free(cad);
}

fclose(archivo);

return 0;
}
